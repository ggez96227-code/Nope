local OrionLib = loadstring(game:HttpGet(('https://raw.githubusercontent.com/jensonhirst/Orion/main/source')))()
local Window = OrionLib:MakeWindow({
	Name = "Script Hub",
	HidePremium = false,
	SaveConfig = true,
	ConfigFolder = "ScriptHub"
})

local MainTab = Window:MakeTab({
	Name = "Main",
	Icon = "rbxassetid://4483345998",
	PremiumOnly = false
})

MainTab:AddLabel("Script Fix Wait")

-- Ví dụ nút Fix Wait
MainTab:AddButton({
	Name = "Script delete Once the fix is complete, you will have a clipboard to copy from.",
	Callback = function()
		local oldWait = wait
		getgenv().wait = function(t)
			return task.wait(t or 0)
		end
		OrionLib:MakeNotification({
			Name = "Fix Wait",
			Content = "Đã chuyển wait() sang task.wait()",
			Time = 3
		})
	end
})

OrionLib:Init()
MainTab:AddButton({
	Name = "Join Discord to get the latest scripts.",
	Callback = function()
		if setclipboard then
			setclipboard("https://discord.gg/5uVYwg8qe")
		elseif toclipboard then
			toclipboard("https://discord.gg/5uVYwg8qe")
		end

		OrionLib:MakeNotification({
			Name = "Thành công",
			Content = "Đã copy link Discord vào clipboard!",
			Image = "rbxassetid://4483345998",
			Time = 3
		})
	end
})
